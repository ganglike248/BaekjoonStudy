<한 줄에 여러 개 입력 받기>

A, B = map(int, input().split())
map: 하나씩 할당
int: 타입 선정
split(): 한 줄에 여러 개 입력 시 한 개씩 분리

A, B = map(int, sys.stdin.readline().split()) # 한 줄에 여러 개 받기

<리스트 입력 받기>

list = [] # 리스트 초기화

list = list(map(int, input().split())) # 리스트 초기화와 동시에 값 한 번에 입력 받기

list.append(int(input())) # 입력 받은 정수를 리스트에 바로 추가
max(list) # 리스트의 값 중 최대값
list.index(max(list)) # 리스트의 값 중 최대값의 인덱스 반환

<큐 사용법>

from collections import deque # 라이브러리에서 큐 가져오기

q = deque(range(1, 11)) # 큐에 1~10 넣기

len(q) # 큐의 길이 확인
q.popleft() # 큐의 가장 첫 번째 요소 반환
q.append(1) # 큐에 1 추가
