q = deque(range(1, 11)) # 큐에 1~10 넣기

len(q) # 큐의 길이 확인
q.popleft() # 큐의 가장 첫 번째 요소 반환

nums = sorted(nums) # 정렬 후 반환

sorted(nums, key=len, reverse=True) # key를 사용해서 특정 조건을 따라 정렬 가능, reverse로 내림차순 가능
key=abs # 절대값 기준으로 정렬
key=len # 문자열 길이를 기준으로 정렬

key=lambda x : (len(x), x) # 튜플을 이용해서 문자열 길이(len(x))로 먼저 정렬 후 오름차순(x) 정렬
# 조건이 2개 이상도 가능

<최대공약수, 최소공배수 구하기>

A와 B라는 수가 있을 때,

최대공약수는 1부터 A * B까지 수 중 A와 B를 동시에 나누었을 때, 둘 다 나머지가 0으로 나누어 떨어지는 가장 큰 값이다.

최소공배수는 두 수를 곱한 값에 최대공약수를 나눈 값이다.
